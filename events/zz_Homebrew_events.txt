country_event = {

	id = homebrew.31
	title = homebrew.31.t
	desc = homebrew.31.d
	hidden = yes
	picture = POPE_PREACHING_eventPicture
	
	is_triggered_only = yes
		
	trigger = {
		NOT = { has_idea = AIG_mining_regulations }
		any_owned_province = {
			OR = {
				has_province_modifier = homebrew_mining_regulations_damestear
				has_province_modifier = homebrew_mining_regulations_relic
			}
		}
	}

	mean_time_to_happen = { days = 1 }

	option = {
		name = "homebrew.31.1"
		clr_country_flag = mining_regulations_start
		every_owned_province = {
			if = {
				limit = { has_province_modifier = homebrew_mining_regulations_damestear }
				remove_province_modifier = homebrew_mining_regulations_damestear
				add_permanent_province_modifier = {
					name = damestear_rich_deposit
					duration = -1
				}
				REB = {
					change_variable = {
						which = homebrew_permadamestear
						value = -1
					}
				}
			}
			if = {
				limit = { has_province_modifier = homebrew_mining_regulations_relic }
				remove_province_modifier = homebrew_mining_regulations_relic
				add_permanent_province_modifier = {
					name = pr_great_ruin
					duration = -1
				}
			}
			if = {
				limit = { has_province_flag = homebrew_permanent_damestear }
				clr_province_flag = homebrew_permanent_damestear
				REB = {
					change_variable = {
						which = homebrew_permadamestear
						value = -1
					}
				}
			}
			if = {
				limit = { has_province_flag = homebrew_permanent_precursor_relics }
				clr_province_flag = homebrew_permanent_precursor_relics
			}
		}
	}
}
